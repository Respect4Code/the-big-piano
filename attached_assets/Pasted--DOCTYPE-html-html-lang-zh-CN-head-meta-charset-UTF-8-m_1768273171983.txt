        // === EVENT LISTENERS ===
        document.addEventListener('DOMContentLoaded', function() {
            // Language Toggle
            const langBtns = document.querySelectorAll('.lang-btn');
            
            langBtns.forEach(btn => {
                btn.addEventListener('click', () => {
                    // Update buttons
                    langBtns.forEach(b => b.classList.remove('active'));
                    btn.classList.add('active');
                    
                    const lang = btn.dataset.lang;
                    
                    // Update body attribute for CSS
                    document.body.dataset.lang = lang;
                    
                    // Update text elements with both data-en and data-zh attributes
                    document.querySelectorAll('[data-en][data-zh]').forEach(el => {
                        el.textContent = el.getAttribute(`data-${lang}`);
                    });
                    
                    // Toggle main dialogue content
                    document.getElementById('english-content').style.display = lang === 'en' ? 'block' : 'none';
                    document.getElementById('chinese-content').style.display = lang === 'zh' ? 'block' : 'none';
                    
                    // Toggle Elephant Poem
                    document.getElementById('elephant-poem-en').style.display = lang === 'en' ? 'block' : 'none';
                    document.getElementById('elephant-poem-zh').style.display = lang === 'zh' ? 'block' : 'none';
                    
                    // Toggle accordion content - FIXED VERSION
                    document.querySelectorAll('.accordion-content [data-en]').forEach(el => {
                        el.style.display = lang === 'en' ? 'block' : 'none';
                    });
                    document.querySelectorAll('.accordion-content [data-zh]').forEach(el => {
                        el.style.display = lang === 'zh' ? 'block' : 'none';
                    });
                    
                    // Save preference
                    localStorage.setItem('preferredLanguage', lang);
                });
            });
            
            // Quick Nav Chips
            document.querySelectorAll('.nav-chip').forEach(chip => {
                chip.addEventListener('click', function() {
                    const targetId = this.dataset.target;
                    
                    // Update active chip
                    document.querySelectorAll('.nav-chip').forEach(c => c.classList.remove('active'));
                    this.classList.add('active');
                    
                    // Scroll to target
                    const target = document.getElementById(targetId);
                    if (target) {
                        target.scrollIntoView({ behavior: 'smooth', block: 'start' });
                    }
                });
            });
            
            // Elephant Gate Click
            document.querySelector('.elephant-question').addEventListener('click', function() {
                const lang = localStorage.getItem('preferredLanguage') || 'en';
                const poem = document.getElementById(`elephant-poem-${lang}`);
                
                if (poem.style.display === 'none' || poem.style.display === '') {
                    poem.style.display = 'block';
                    setTimeout(() => {
                        poem.scrollIntoView({ behavior: 'smooth', block: 'center' });
                    }, 100);
                } else {
                    poem.style.display = 'none';
                }
            });
            
            // Accordion Functionality
            document.querySelectorAll('.accordion-header').forEach(header => {
                header.addEventListener('click', function() {
                    const content = this.nextElementSibling;
                    const icon = this.querySelector('.accordion-icon');
                    
                    // Toggle current
                    content.classList.toggle('active');
                    icon.textContent = content.classList.contains('active') ? 'â–²' : 'â–¼';

                    document.querySelectorAll('.accordion-content').forEach(otherContent => {
                        if (otherContent !== content && otherContent.classList.contains('active')) {
                            otherContent.classList.remove('active');
                            otherContent.previousElementSibling.querySelector('.accordion-icon').textContent = 'â–¼';
                        }
                    });
                });
            });

            const savedLang = localStorage.getItem('preferredLanguage') || 'en';
