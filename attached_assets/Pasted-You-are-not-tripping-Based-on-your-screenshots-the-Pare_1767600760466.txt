      blurb.textContent =
        'A three-year-old discovers the magic of a real piano.\n' +
        'From electronic plinks to the grand sound of a family heirloom,\n' +
        'it is a story about music, legacy, and pure joy.';
    }
  }

  btnEn.addEventListener('click', () => setLang('en'));
  btnZh.addEventListener('click', () => setLang('zh'));

  // ---------- PIN Gate (local-only) ----------
  const gate = document.getElementById('gate');
  const locked = document.getElementById('lockedContent');
  const pinInput = document.getElementById('pinInput');
  const pinAction = document.getElementById('pinAction');
  const gateTitle = document.getElementById('gateTitle');
  const gateHint = document.getElementById('gateHint');
  const gateMsg = document.getElementById('gateMsg');
  const forgotBtn = document.getElementById('forgotBtn');

  const PIN_KEY = 'parentPin';

  function hasPin(){
    return !!localStorage.getItem(PIN_KEY);
  }

  function showMsg(text){
    gateMsg.textContent = text || '';
  }

  function lock(){
    locked.style.display = 'none';
    gate.style.display = 'block';
    pinInput.value = '';
    showMsg('');
    forgotBtn.style.display = hasPin() ? 'inline' : 'none';

    if(!hasPin()){
      gateTitle.textContent = 'Set a Parent PIN';
      pinInput.placeholder = 'Set new PIN';
      pinAction.textContent = 'Set PIN';
      gateHint.textContent = 'Choose a PIN you will remember. Stored only on this device.';
    } else {
      gateTitle.textContent = 'Parent access';
      pinInput.placeholder = 'Enter PIN';
      pinAction.textContent = 'Unlock';
      gateHint.textContent = 'This space is for adult thinking. The PIN is stored only on this device.';
    }
  }

  function unlock(){
    gate.style.display = 'none';
    locked.style.display = 'block';
  }

  pinAction.addEventListener('click', () => {
    const v = (pinInput.value || '').trim();
    if(v.length < 3){
      showMsg('PIN is too short.');
      return;
    }

    if(!hasPin()){
      localStorage.setItem(PIN_KEY, v);
      showMsg('PIN set. Unlockingâ€¦');
      setTimeout(unlock, 150);
      return;
    }

    const saved = localStorage.getItem(PIN_KEY);
    if(v === saved){
      unlock();
    } else {
      showMsg('Incorrect PIN.');
      forgotBtn.style.display = 'inline';
    }
  });

  forgotBtn.addEventListener('click', () => {
    const ok = confirm('Forgot PIN? This will reset Parent access on this device. You can set a new PIN afterwards.');
    if(!ok) return;
    localStorage.removeItem(PIN_KEY);
    lock();
    showMsg('PIN reset. Please set a new PIN.');
  });

  // ---------- Init ----------
  setLang(localStorage.getItem('parentLang') || 'en');
  lock();
</script>
