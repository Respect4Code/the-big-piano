You’ve already done the hard part: you’ve *nailed the product surface* (story → one question → local timeline) and you’ve protected the “resonance” by keeping the child layer clean.

Below is **Replit-ready, single-file** code (`index.html`) that implements exactly what your screenshots show:

* English / 中文 toggle
* Story pane (left) + Timeline pane (right)
* One Question section with:

  * **Record child** (audio stays local; stored in **IndexedDB** on-device)
  * **Write one line** (parent types exact words)
  * **Optional icons** (tap to select)
  * Save to timeline (stored locally; **no uploads, no tracking**)
* Timeline list with playback of saved audio (if recorded)
* Export notes (JSON) **without audio** (privacy-first)
* Wipe local data (notes + audio)
* Parent toggle (top-right) showing:

  * Observation prompts (non-leading guidance)
  * Contrarian prompts (parent-only)
  * Bridge layer (age-gated placeholder) to hold your ivory→Bitcoin continuum later

const LS_KEY = "bigpiano_timeline_v1";
const DB_NAME = "bigpiano_audio_v1";
const DB_STORE = "clips";

function uid(){
  return Math.random().toString(16).slice(2) + "-" + Date.now().toString(16);
}

function loadTimeline(){
  try{
    const raw = localStorage.getItem(LS_KEY);
    return raw ? JSON.parse(raw) : [];
  }catch{
    return [];
  }
}

function saveTimeline(list){
  localStorage.setItem(LS_KEY, JSON.stringify(list));
}

function download(filename, text){
  const a = document.createElement("a");
  a.href = URL.createObjectURL(new Blob([text], {type:"application/json"}));
  a.download = filename;
  document.body.appendChild(a);
  a.click();
  a.remove();
}

/* IndexedDB helpers */
function openDB(){
  return new Promise((resolve, reject) => {
    const req = indexedDB.open(DB_NAME, 1);
    req.onupgradeneeded = () => {
      const db = req.result;
      if(!db.objectStoreNames.contains(DB_STORE)){
        db.createObjectStore(DB_STORE);
      }
    };
    req.onsuccess = () => resolve(req.result);
    req.onerror = () => reject(req.error);
  });
}

async function putAudio(id, blob){
  const db = await openDB();
  return new Promise((resolve, reject) => {
    const tx = db.transaction(DB_STORE, "readwrite");
    tx.objectStore(DB_STORE).put(blob, id);
    tx.oncomplete = () => resolve();
    tx.onerror = () => reject(tx.error);
  });
}

async function getAudio(id){
  const db = await openDB();
  return new Promise((resolve, reject) => {
    const tx = db.transaction(DB_STORE, "readonly");
    const req = tx.objectStore(DB_STORE).get(id);
    req.onsuccess = () => resolve(req.result || null);
    req.onerror = () => reject(req.error);
  });
}

async function delAudio(id){
  const db = await openDB();
  return new Promise((resolve, reject) => {
    const tx = db.transaction(DB_STORE, "readwrite");
    tx.objectStore(DB_STORE).delete(id);
    tx.oncomplete = () => resolve();
    tx.onerror = () => reject(tx.error);
  });
}

async function wipeAllAudio(){
  const db = await openDB();
  return new Promise((resolve, reject) => {
    const tx = db.transaction(DB_STORE, "readwrite");
    tx.objectStore(DB_STORE).clear();
    tx.oncomplete = () => resolve();
    tx.onerror = () => reject(tx.error);
  });
}

async function renderTimeline(){
  const body = $("timelineBody");
  if(!timeline.length){
    body.innerHTML = `<div class="empty">${escapeHtml(STORY[LANG].empty)}</div>`;
    return;
  }

  // Newest first
  const sorted = [...timeline].sort((a,b) => b.ts - a.ts);

  body.innerHTML = "";
  for(const item of sorted){
    const el = document.createElement("div");
    el.className = "entry";

    const date = new Date(item.ts);
    const dateStr = date.toLocaleString(undefined, {year:"numeric", month:"short", day:"2-digit", hour:"2-digit", minute:"2-digit"});

    const iconsHtml = (item.icons && item.icons.length)
      ? `<div class="entryChips">${item.icons.map(i => `<span class="miniChip">${escapeHtml(i)}</span>`).join("")}</div>`
      : "";

    const audioId = item.audioId || null;

    el.innerHTML = `
      <div class="entryHead">
        <div class="entryMeta">
          <span>${escapeHtml(dateStr)}</span>
          ${item.age ? `<span>• ${escapeHtml(item.age)}</span>` : ""}
        </div>
        <button class="smallBtn" data-del="${escapeHtml(item.id)}" type="button">Delete</button>
      </div>

      ${item.line ? `<p class="entryLine">${escapeHtml(item.line)}</p>` : `<p class="entryLine" style="color: rgba(255,255,255,.55)">(no note)</p>`}

      ${iconsHtml}

      <div class="audioRow" id="audioRow_${escapeHtml(item.id)}">
        ${audioId ? `<span class="badge">Audio</span>` : `<span class="badge">No audio</span>`}
      </div>
    `;

    body.appendChild(el);

    // Delete handler
    el.querySelector(`[data-del="${item.id}"]`).onclick = async () => {
      if(!confirm("Delete this moment?")) return;
      // remove audio if present
      if(item.audioId) await delAudio(item.audioId);
      timeline = timeline.filter(x => x.id !== item.id);
      saveTimeline(timeline);
      renderTimeline();
    };

    // Load and attach audio player if exists
    if(audioId){
      const row = el.querySelector(`#audioRow_${CSS.escape(item.id)}`);
      const blob = await getAudio(audioId);
      if(blob){
        const url = URL.createObjectURL(blob);
        const audio = document.createElement("audio");
        audio.controls = true;
        audio.src = url;
        row.appendChild(audio);
      }else{
        const missing = document.createElement("span");
        missing.className = "badge";
        missing.textContent = "Audio missing";
        row.appendChild(missing);
      }
    }
  }
}

function escapeHtml(str){
  return String(str)
    .replaceAll("&","&amp;")
    .replaceAll("<","&lt;")
    .replaceAll(">","&gt;")
    .replaceAll('"',"&quot;")
    .replaceAll("'","&#039;");
}

async function startRecording(){
  try{
    const stream = await navigator.mediaDevices.getUserMedia({ audio: true });
    recordedChunks = [];
    pendingAudioBlob = null;

    mediaRecorder = new MediaRecorder(stream);
    mediaRecorder.ondataavailable = (e) => {
      if(e.data && e.data.size > 0) recordedChunks.push(e.data);
    };
    mediaRecorder.onstop = () => {
      // stop tracks
      stream.getTracks().forEach(t => t.stop());
      pendingAudioBlob = new Blob(recordedChunks, { type: "audio/webm" });

      const url = URL.createObjectURL(pendingAudioBlob);
      $("recPreview").src = url;
      $("recPreviewRow").style.display = "flex";

      $("recStatus").textContent = STORY[LANG].recStatusIdle;
    };

    mediaRecorder.start();
    $("btnRecStart").disabled = true;
    $("btnRecStop").disabled = false;
    $("recStatus").textContent = STORY[LANG].recStatusOn;
    $("recPreviewRow").style.display = "none";
  }catch(err){
    alert("Microphone permission is required to record audio.");
    console.error(err);
  }
}

function stopRecording(){
  if(mediaRecorder && mediaRecorder.state !== "inactive"){
    mediaRecorder.stop();
  }
  $("btnRecStart").disabled = false;
  $("btnRecStop").disabled = true;
}

  const entry = {
    id,
    ts: Date.now(),
    age: age || "",
    line: line || "",
    icons,
    audioId
  };

  timeline.push(entry);
  saveTimeline(timeline);

  // Reset input state (keep story untouched)
  $("childAge").value = "";
  $("oneLine").value = "";
  selectedIcons.clear();
  renderIcons();

  pendingAudioBlob = null;
  $("recPreviewRow").style.display = "none";
  $("recPreview").src = "";

  renderTimeline();
}

/* =========================
   Export / Wipe
   ========================= */
function exportNotes(){
  const payload = {
    app: "The Big Piano",
    version: 1,
    exportedAt: new Date().toISOString(),
    notes: timeline
      .map(({id, ts, age, line, icons}) => ({ id, ts, age, line, icons }))
      .sort((a,b)=>a.ts-b.ts)
  };
  download("big-piano-notes.json", JSON.stringify(payload, null, 2));
}

async function wipeAll(){
  if(!confirm("This will delete all local notes and recordings from this device. Continue?")) return;
  timeline = [];
  saveTimeline(timeline);
  await wipeAllAudio();
  renderTimeline();

  $("childAge").value = "";
  $("oneLine").value = "";
  selectedIcons.clear();
  renderIcons();
  pendingAudioBlob = null;
  $("recPreviewRow").style.display = "none";
  $("recPreview").src = "";
}

  return localStorage.getItem("bigpiano_bridge_unlocked_v1") === "1";
}

  localStorage.setItem("bigpiano_bridge_unlocked_v1", on ? "1" : "0");
}

function refreshBridgeUI(){
  const unlocked = isBridgeUnlocked();
  $("bridgeContent").style.display = unlocked ? "block" : "none";
  $("gateBadge").textContent = unlocked ? STORY[LANG].unlocked : STORY[LANG].locked;

function unlockBridge(){
  const raw = $("gateAge").value.trim();
  const n = Number(raw);
  if(!raw || !Number.isFinite(n)){
    alert("Enter an age number (e.g., 10).");
    return;
  }
  // Simple gate: 8+
  if(n < 8){
    alert("Bridge is intended for later (8+).");
    return;
  }
  setBridgeUnlocked(true);
  refreshBridgeUI();
}
